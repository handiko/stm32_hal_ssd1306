## 3. main.c 

__stm32f411_fw_ssd1306/Core/Src/main.c__

```c
#include "ssd1306.h"

int main(void)
{
    ssd1306_init();

    ssd1306_write_string(font6x8, "ABC");
    ssd1306_enter();

    ssd1306_write_string(font7x10, "ABC");
    ssd1306_enter();

    ssd1306_write_string(font11x18, "ABC");
    ssd1306_enter();

    ssd1306_write_string(font16x26, "ABC");
    ssd1306_enter();

    ssd1306_update_screen();
}

```
